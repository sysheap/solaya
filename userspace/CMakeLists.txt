# userspace/CMakeLists.txt — build userspace rust binaries.
#
# Outputs land in ${SOLAYA_USERSPACE_ARTIFACT_DIR} (build/userspace/artifacts
# by default).  Buildroot stages these into its rootfs overlay under /bin/;
# standard POSIX commands (cat, ls, echo, ...) come from GNU coreutils in
# buildroot, not from a Rust reimplementation.

cmake_minimum_required(VERSION 3.25)
project(solaya_userspace LANGUAGES NONE)

set(SOLAYA_USERSPACE_ARTIFACT_DIR "${CMAKE_BINARY_DIR}/userspace/artifacts"
    CACHE PATH "Where userspace binaries are deposited for kernel embedding."
)

# Rust target for userspace binaries. Hardcoded today (riscv64 only).
set(_rust_target "riscv64gc-unknown-linux-musl")

# The userspace linker resolves `riscv64-linux-musl-clang` by name (via
# userspace/.cargo/config.toml [target.*] linker = ...); prepend
# ${SOLAYA_CROSS_BIN} (which holds the clang wrapper scripts generated by
# cmake/clang_wrapper.cmake) to PATH for every cargo invocation below.
#
# PATH is extended via `sh -c` so $PATH is read at build time, not CMake
# configure time.  Freezing $ENV{PATH} into build.ninja breaks when the same
# build/ dir is shared between the rootful podman devcontainer and the host
# (the host sees /root/.cargo/bin from the container, hits EACCES, and
# aborts with "permission denied").

# -----------------------------------------------------------------------------
# userspace-rust: `cargo build --bins` for the workspace at userspace/.
# -----------------------------------------------------------------------------
add_custom_target(userspace-rust
    COMMAND ${CMAKE_COMMAND} -E make_directory "${SOLAYA_USERSPACE_ARTIFACT_DIR}"
    COMMAND sh -c "PATH=\"${SOLAYA_CROSS_BIN}:$PATH\" \
CARGO_TARGET_DIR=\"${CMAKE_BINARY_DIR}/userspace/target\" \
exec ${SOLAYA_CARGO} build \
--bins --release \
--target ${_rust_target} \
--artifact-dir \"${SOLAYA_USERSPACE_ARTIFACT_DIR}\" \
-Z unstable-options"
    # Run from userspace/ so cargo picks up userspace/.cargo/config.toml
    # (which sets the riscv64 linker).  With --manifest-path from a parent
    # dir cargo finds only the parent's .cargo/config.toml and the linker
    # override is lost.
    WORKING_DIRECTORY "${CMAKE_SOURCE_DIR}/userspace"
    DEPENDS headers-generated compiler-rt-builtins
    USES_TERMINAL
    VERBATIM
    COMMENT "Building userspace Rust binaries (${_rust_target})"
)

# -----------------------------------------------------------------------------
# Aggregate: everything a kernel build needs from userspace.
# -----------------------------------------------------------------------------
add_custom_target(userspace-all
    DEPENDS userspace-rust dash doom
    COMMENT "Userspace artifact set (rust bins + dash/sh + doom)"
)
